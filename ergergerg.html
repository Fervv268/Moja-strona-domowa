Pełny kod skryptu create-github-pages.sh

#\!/usr/bin/env bash
# Automatyczne zakładanie i publikacja strony GitHub Pages
# Autor: ChatGPT – 2025-06-26

set -euo pipefail

################ KONFIGURACJA ################
TITLE="${TITLE:-Moja strona domowa}"
DESCRIPTION="${DESCRIPTION:-Witamy na mojej stronie!}"
BRANCH="main"
THEME="jekyll-theme-minimal"
##############################################

# 0. Sprawdzenie zależności
for cmd in gh git; do
  command -v "$cmd" &>/dev/null || {
    echo "❌  Brak '$cmd'. Zainstaluj i spróbuj ponownie."; exit 1; }
done

# 1. Nazwa użytkownika
USERNAME="$(gh api user --jq .login)"
REPO="${USERNAME}.github.io"

# 2. Tworzenie repozytorium, jeśli nie istnieje
if ! gh repo list "$USERNAME" --json name -q '.[].name' | grep -qx "$REPO"; then
  gh repo create "$REPO" --public -y \
     --description "$TITLE" \
     --homepage "https://${USERNAME}.github.io"
fi

# 3. Katalog roboczy
WORKDIR="$(mktemp -d)"
git clone "https://github.com/${USERNAME}/${REPO}.git" "$WORKDIR"
cd "$WORKDIR"

# 4. README.md
echo "# ${TITLE}" > README.md
git add README.md && git commit -m "Add README"

# 5. _config.yml
cat > _config.yml </dev/null
{
  "source": { "branch": "$BRANCH", "path": "/" }
}
EOF

echo
echo "✅  Strona będzie dostępna wkrótce pod:"
echo "   https://${USERNAME}.github.io"
echo "(Katalog roboczy: $WORKDIR)"
